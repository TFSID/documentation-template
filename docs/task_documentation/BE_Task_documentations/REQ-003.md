# REQ-003: API Endpoint Paket Request

## Deskripsi Tugas
Mengimplementasikan API endpoints yang komprehensif untuk melakukan operasi CRUD (Create, Read, Update, Delete) pada paket request, termasuk validasi data, error handling, dan response formatting yang konsisten.

## Tujuan
- Menyediakan API yang robust dan scalable untuk manajemen paket request
- Memastikan data consistency melalui proper validation
- Memberikan response yang consistent dan informatif
- Mengoptimalkan performance untuk handling volume request yang tinggi

## Deliverables
- RESTful API endpoints untuk paket request operations
- Comprehensive input validation dan sanitization
- Error handling dengan proper HTTP status codes
- API documentation dengan examples
- Rate limiting dan security measures

## Step-by-Step Implementasi

### 1. API Architecture Design
```
API Structure:
├── /api/v1/package-requests
│   ├── GET    /                     # List all requests
│   ├── POST   /                     # Create new request
│   ├── GET    /:id                  # Get specific request
│   ├── PUT    /:id                  # Update request
│   ├── DELETE /:id                  # Delete request
│   ├── POST   /:id/submit           # Submit request
│   ├── PUT    /:id/status           # Update status
│   └── GET    /:id/history          # Get status history
├── /api/v1/package-requests/:id/documents
│   ├── GET    /                     # List documents
│   ├── POST   /upload               # Upload document
│   ├── DELETE /:documentId          # Delete document
│   └── PUT    /:documentId/verify   # Verify document
└── /api/v1/package-requests/stats   # Statistics endpoint
```

```javascript
// controllers/packageRequestController.js
const PackageRequestService = require('../services/PackageRequestService');
const logger = require('../utils/logger');

class PackageRequestController {
  // GET /api/v1/package-requests
  async getAllRequests(req, res) {
    try {
      const {
        page = 1,
        limit = 10,
        status,
        priority,
        assigned_to,
        date_from,
        date_to,
        search
      } = req.query;

      const filters = {
        user_id: req.user.id, // Users can only see their own requests unless admin
        ...(status && { status }),
        ...(priority && { priority }),
        ...(assigned_to && { assigned_to }),
        ...(date_from && { date_from }),
        ...(date_to && { date_to }),
        ...(search && { search })
      };

      // Check if user is admin/staff to see all requests
      if (req.user.role === 'admin' || req.user.role === 'staff') {
        delete filters.user_id;
      }

      const result = await PackageRequestService.getAllRequests({
        filters,
        pagination: { page: parseInt(page), limit: parseInt(limit) },
        sorting: { created_at: 'desc' }
      });

      res.json({
        success: true,
        data: result.data,
        pagination: result.pagination,
        message: 'Package requests retrieved successfully'
      });
    } catch (error) {
      logger.error('Error getting package requests:', error);
      res.status(500).json({
        success: false,
        error: {
          code: 'INTERNAL_SERVER_ERROR',
          message: 'Failed to retrieve package requests'
        }
      });
    }
  }

  // POST /api/v1/package-requests
  async createRequest(req, res) {
    try {
      const requestData = {
        ...req.body,
        user_id: req.user.id,
        created_by: req.user.id
      };

      const packageRequest = await PackageRequestService.createRequest(requestData);

      res.status(201).json({
        success: true,
        data: packageRequest,
        message: 'Package request created successfully'
      });
    } catch (error) {
      logger.error('Error creating package request:', error);
      
      if (error.name === 'ValidationError') {
        return res.status(400).json({
          success: false,
          error: {
            code: 'VALIDATION_ERROR',
            message: 'Invalid input data',
            details: error.details
          }
        });
      }

      res.status(500).json({
        success: false,
        error: {
          code: 'INTERNAL_SERVER_ERROR',
          message: 'Failed to create package request'
        }
      });
    }
  }

  // GET /api/v1/package-requests/:id
  async getRequestById(req, res) {
    try {
      const { id } = req.params;
      
      // Check if user has access to this request
      const hasAccess = await PackageRequestService.checkUserAccess(id, req.user.id);
      if (!hasAccess && req.user.role !== 'admin' && req.user.role !== 'staff') {
        return res.status(403).json({
          success: false,
          error: {
            code: 'ACCESS_DENIED',
            message: 'You do not have permission to access this request'
          }
        });
      }

      const packageRequest = await PackageRequestService.getRequestById(id);
      
      if (!packageRequest) {
        return res.status(404).json({
          success: false,
          error: {
            code: 'NOT_FOUND',
            message: 'Package request not found'
          }
        });
      }

      res.json({
        success: true,
        data: packageRequest,
        message: 'Package request retrieved successfully'
      });
    } catch (error) {
      logger.error('Error getting package request:', error);
      res.status(500).json({
        success: false,
        error: {
          code: 'INTERNAL_SERVER_ERROR',
          message: 'Failed to retrieve package request'
        }
      });
    }
  }

  // PUT /api/v1/package-requests/:id
  async updateRequest(req, res) {
    try {
      const { id } = req.params;
      const updateData = {
        ...req.body,
        updated_by: req.user.id
      };

      // Check if user can update this request
      const canUpdate = await PackageRequestService.canUserUpdate(id, req.user.id);
      if (!canUpdate) {
        return res.status(403).json({
          success: false,
          error: {
            code: 'ACCESS_DENIED',
            message: 'You do not have permission to update this request'
          }
        });
      }

      const updatedRequest = await PackageRequestService.updateRequest(id, updateData);

      res.json({
        success: true,
        data: updatedRequest,
        message: 'Package request updated successfully'
      });
    } catch (error) {
      logger.error('Error updating package request:', error);
      
      if (error.name === 'ValidationError') {
        return res.status(400).json({
          success: false,
          error: {
            code: 'VALIDATION_ERROR',
            message: 'Invalid input data',
            details: error.details
          }
        });
      }

      res.status(500).json({
        success: false,
        error: {
          code: 'INTERNAL_SERVER_ERROR',
          message: 'Failed to update package request'
        }
      });
    }
  }

  // POST /api/v1/package-requests/:id/submit
  async submitRequest(req, res) {
    try {
      const { id } = req.params;

      // Check if user can submit this request
      const canSubmit = await PackageRequestService.canUserSubmit(id, req.user.id);
      if (!canSubmit) {
        return res.status(403).json({
          success: false,
          error: {
            code: 'ACCESS_DENIED',
            message: 'You cannot submit this request'
          }
        });
      }

      const submittedRequest = await PackageRequestService.submitRequest(id, req.user.id);

      res.json({
        success: true,
        data: submittedRequest,
        message: 'Package request submitted successfully'
      });
    } catch (error) {
      logger.error('Error submitting package request:', error);
      res.status(500).json({
        success: false,
        error: {
          code: 'INTERNAL_SERVER_ERROR',
          message: 'Failed to submit package request'
        }
      });
    }
  }

  // PUT /api/v1/package-requests/:id/status
  async updateStatus(req, res) {
    try {
      const { id } = req.params;
      const { status, reason } = req.body;

      const updatedRequest = await PackageRequestService.updateStatus(
        id, 
        status, 
        req.user.id, 
        reason
      );

      res.json({
        success: true,
        data: updatedRequest,
        message: 'Request status updated successfully'
      });
    } catch (error) {
      logger.error('Error updating request status:', error);
      res.status(500).json({
        success: false,
        error: {
          code: 'INTERNAL_SERVER_ERROR',
          message: 'Failed to update request status'
        }
      });
    }
  }
}

module.exports = new PackageRequestController();
```

#### 3.3 Service Layer Implementation
```javascript
// services/PackageRequestService.js
const PackageRequestRepository = require('../repositories/PackageRequestRepository');
const ValidationService = require('./ValidationService');
const NotificationService = require('./NotificationService');
const logger = require('../utils/logger');

class PackageRequestService {
  async createRequest(requestData) {
    try {
      // Validate request data
      const validationResult = await ValidationService.validatePackageRequest(requestData);
      if (!validationResult.isValid) {
        throw new ValidationError('Invalid request data', validationResult.errors);
      }

      // Generate request number
      requestData.request_number = await this.generateRequestNumber();

      // Create the request
      const packageRequest = await PackageRequestRepository.create(requestData);

      // Create related records (personal info, preferences, etc.)
      await this.createRelatedRecords(packageRequest.id, requestData);

      // Send notification
      await NotificationService.sendRequestCreatedNotification(packageRequest);

      logger.info(`Package request created: ${packageRequest.id} by user ${requestData.user_id}`);

      return packageRequest;
    } catch (error) {
      logger.error('Error creating package request:', error);
      throw error;
    }
  }

  async getAllRequests(filters, options) {
    try {
      const result = await PackageRequestRepository.findAll(filters, options);
      return result;
    } catch (error) {
      logger.error('Error getting package requests:', error);
      throw error;
    }
  }

  async getRequestById(id) {
    try {
      const packageRequest = await PackageRequestRepository.findById(id);
      
      if (!packageRequest) {
        return null;
      }

      // Load related data
      packageRequest.personal_info = await PackageRequestRepository.getPersonalInfo(id);
      packageRequest.travel_preferences = await PackageRequestRepository.getTravelPreferences(id);
      packageRequest.budget_info = await PackageRequestRepository.getBudgetInfo(id);
      packageRequest.documents = await PackageRequestRepository.getDocuments(id);

      return packageRequest;
    } catch (error) {
      logger.error('Error getting package request by ID:', error);
      throw error;
    }
  }

  async updateRequest(id, updateData) {
    try {
      // Validate update data
      const validationResult = await ValidationService.validateUpdateRequest(updateData);
      if (!validationResult.isValid) {
        throw new ValidationError('Invalid update data', validationResult.errors);
      }

      const updatedRequest = await PackageRequestRepository.update(id, updateData);

      // Update related records if needed
      if (updateData.personal_info) {
        await PackageRequestRepository.updatePersonalInfo(id, updateData.personal_info);
      }

      if (updateData.travel_preferences) {
        await PackageRequestRepository.updateTravelPreferences(id, updateData.travel_preferences);
      }

      if (updateData.budget_info) {
        await PackageRequestRepository.updateBudgetInfo(id, updateData.budget_info);
      }

      logger.info(`Package request updated: ${id}`);

      return updatedRequest;
    } catch (error) {
      logger.error('Error updating package request:', error);
      throw error;
    }
  }

  async submitRequest(id, userId) {
    try {
      const packageRequest = await PackageRequestRepository.findById(id);
      
      if (!packageRequest) {
        throw new NotFoundError('Package request not found');
      }

      if (packageRequest.user_id !== userId) {
        throw new ForbiddenError('You can only submit your own requests');
      }

      if (packageRequest.status !== 'draft') {
        throw new BadRequestError('Only draft requests can be submitted');
      }

      // Update status to submitted
      const submittedRequest = await PackageRequestRepository.update(id, {
        status: 'submitted',
        submitted_at: new Date(),
        submitted_by: userId
      });

      // Log status change
      await PackageRequestRepository.logStatusChange(id, 'draft', 'submitted', userId);

      // Send notifications
      await NotificationService.sendRequestSubmittedNotification(submittedRequest);

      logger.info(`Package request submitted: ${id}`);

      return submittedRequest;
    } catch (error) {
      logger.error('Error submitting package request:', error);
      throw error;
    }
  }

  async updateStatus(id, newStatus, userId, reason) {
    try {
      const packageRequest = await PackageRequestRepository.findById(id);
      
      if (!packageRequest) {
        throw new NotFoundError('Package request not found');
      }

      const oldStatus = packageRequest.status;
      const updatedRequest = await PackageRequestRepository.update(id, {
        status: newStatus,
        updated_at: new Date(),
        updated_by: userId,
        ...(reason && { internal_notes: reason })
      });

      // Log status change
      await PackageRequestRepository.logStatusChange(id, oldStatus, newStatus, userId, reason);

      // Send notifications based on new status
      await NotificationService.sendStatusChangeNotification(updatedRequest, oldStatus, newStatus);

      logger.info(`Request status changed: ${id} from ${oldStatus} to ${newStatus} by user ${userId}`);

      return updatedRequest;
    } catch (error) {
      logger.error('Error updating request status:', error);
      throw error;
    }
  }

  // Helper methods
  async generateRequestNumber() {
    const year = new Date().getFullYear();
    const month = String(new Date().getMonth() + 1).padStart(2, '0');
    const count = await PackageRequestRepository.getMonthlyCount(year, month);
    const sequence = String(count + 1).padStart(4, '0');
    
    return `REQ-${year}${month}-${sequence}`;
  }

  async checkUserAccess(requestId, userId) {
    const request = await PackageRequestRepository.findById(requestId);
    return request && (request.user_id === userId);
  }

  async canUserUpdate(requestId, userId) {
    const request = await PackageRequestRepository.findById(requestId);
    return request && request.user_id === userId && request.status === 'draft';
  }

  async canUserSubmit(requestId, userId) {
    const request = await PackageRequestRepository.findById(requestId);
    return request && request.user_id === userId && request.status === 'draft';
  }
}

module.exports = new PackageRequestService();
```

```javascript
// middleware/validation.js
const { body, param, query, validationResult } = require('express-validator');

const validateRequest = (validationType) => {
  switch (validationType) {
    case 'createPackageRequest':
      return [
        body('personal_info.full_name').notEmpty().isLength({ max: 255 }),
        body('personal_info.nik').isLength({ min: 16, max: 16 }).isNumeric(),
        body('personal_info.birth_date').isISO8601().toDate(),
        body('personal_info.phone_primary').isMobilePhone('id-ID'),
        body('personal_info.email').isEmail(),
        body('travel_preferences.number_of_pilgrims').isInt({ min: 1, max: 50 }),
        body('travel_preferences.departure_date').isISO8601().toDate().isAfter(),
        body('travel_preferences.departure_city').notEmpty().isLength({ max: 100 }),
        body('budget_info.max_budget').isFloat({ min: 0 })
      ];
    
    case 'updateRequestStatus':
      return [
        param('id').isUUID(),
        body('status').isIn(['draft', 'submitted', 'under_review', 'quoted', 'approved', 'rejected']),
        body('reason').optional().isLength({ max: 500 })
      ];
    
    default:
      return [];
  }
};

const validateRequest = (validationType) => {
  const rules = getValidationRules(validationType);
  return async (req, res, next) => {
    await Promise.all(rules.map(validation => validation.run(req)));
    
    const errors = validationResult(req);
    if (!errors.isEmpty()) {
      return res.status(400).json({
        success: false,
        error: {
          code: 'VALIDATION_ERROR',
          message: 'Invalid input data',
          details: errors.array()
        }
      });
    }
    
    next();
  };
};

module.exports = { validateRequest };
```

#### 4.2 Authentication & Authorization Middleware
```javascript
// middleware/auth.js
const jwt = require('jsonwebtoken');
const UserRepository = require('../repositories/UserRepository');

const authenticateToken = async (req, res, next) => {
  try {
    const authHeader = req.headers['authorization'];
    const token = authHeader && authHeader.split(' ')[1];

    if (!token) {
      return res.status(401).json({
        success: false,
        error: {
          code: 'UNAUTHORIZED',
          message: 'Access token required'
        }
      });
    }

    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    const user = await UserRepository.findById(decoded.userId);
    
    if (!user) {
      return res.status(401).json({
        success: false,
        error: {
          code: 'UNAUTHORIZED',
          message: 'Invalid access token'
        }
      });
    }

    req.user = user;
    next();
  } catch (error) {
    return res.status(403).json({
      success: false,
      error: {
        code: 'FORBIDDEN',
        message: 'Invalid or expired token'
      }
    });
  }
};

module.exports = { authenticateToken };
```

### 5. Error Handling
```javascript
// middleware/errorHandler.js
const logger = require('../utils/logger');

const errorHandler = (err, req, res, next) => {
  logger.error('API Error:', {
    message: err.message,
    stack: err.stack,
    url: req.url,
    method: req.method,
    body: req.body,
    params: req.params,
    query: req.query
  });

  // Default error
  let error = {
    code: 'INTERNAL_SERVER_ERROR',
    message: 'An unexpected error occurred'
  };

  // Handle specific error types
  if (err.name === 'ValidationError') {
    error = {
      code: 'VALIDATION_ERROR',
      message: 'Invalid input data',
      details: err.details
    };
  } else if (err.name === 'NotFoundError') {
    error = {
      code: 'NOT_FOUND',
      message: err.message
    };
  } else if (err.name === 'ForbiddenError') {
    error = {
      code: 'FORBIDDEN',
      message: err.message
    };
  } else if (err.name === 'BadRequestError') {
    error = {
      code: 'BAD_REQUEST',
      message: err.message
    };
  }

  const statusCode = {
    'VALIDATION_ERROR': 400,
    'NOT_FOUND': 404,
    'FORBIDDEN': 403,
    'BAD_REQUEST': 400,
    'UNAUTHORIZED': 401,
    'INTERNAL_SERVER_ERROR': 500
  }[error.code] || 500;

  res.status(statusCode).json({
    success: false,
    error,
    timestamp: new Date().toISOString(),
    requestId: req.id
  });
};

module.exports = errorHandler;
```
